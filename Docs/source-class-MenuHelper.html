259:     /**
260:      * Generates and returns a menu for an action.
261:      * @param string $name Name of the action for which to generate the menu
262:      * @param string $type Type of menu (optional, `ul`, `collapse` or `dropdown`)
263:      * @return mixed Html
264:      */
265:     public function get($name, $type = NULL) {
266:         //Dynamic call to the method that generates the requested menu
267:         $name = sprintf('_%s', $name);
268:         
269:         //Checks if the method exists
270:         if(!method_exists($class = get_called_class(), $name))
271:             throw new InternalErrorException(__d('me_cms', 'The %s method does not exist', sprintf('%s::%s()', $class, $name)));
272:         
273:         $menu = $class::$name($type);
274:         
275:         if(empty($menu))
276:             return FALSE;
277:         
278:         //Switch the type of menu
279:         switch($type) {
280:             case 'ul':
281:                 $menu = $this->ul($menu);
282:                 break;
283:             case 'collapse':
284:                 $menu = $this->div('panel', $menu);
285:                 break;
286:         }
287:         
288:         return $menu;
289:     }
