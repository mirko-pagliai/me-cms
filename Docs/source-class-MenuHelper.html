 47:     /**

 52:      * @uses DropdownHelper::dropdown()
 53:      * @uses DropdownHelper::link()
 54:      * @uses AuthHelper::isAdmin()
 55:      */

 57:         //Only admins can access these controllers
 58:         if(!$this->Auth->isAdmin())
 59:             return array();
 60:         
 61:         $menu = array(
 62:             $this->link(__d('me_cms', 'List banners'),      array('controller' => 'banners', 'action' => 'index')),
 63:             $this->link(__d('me_cms', 'Add banner'),        array('controller' => 'banners', 'action' => 'add')),

 66:         );
 67:         
 68:         if($type == 'dropdown')
 69:             return $this->Dropdown->link(__d('me_cms', 'Banners'), array('icon' => 'dollar')).PHP_EOL.$this->Dropdown->dropdown($menu);

207:     /**
208:      * Generates and returns a menu for an action.
209:      * @param string $name Name of the action for which to generate the menu
210:      * @param string $type Type of menu (optional, `ul`, `nav` or `dropdown`)
211:      * @return mixed Menu
212:      * @uses _pages() to generate the menu for "pages" actions
213:      * @uses _photos() to generate the menu for "photos" actions
214:      * @uses _posts() to generate the menu for "posts" actions
215:      * @uses _users() to generate the menu for "users" actions
216:      */
217:     public function get($name, $type = NULL) {
218:         //Dynamic call to the method that generates the requested menu
219:         $name = sprintf('_%s', $name);
220:         $menu = $this->$name($type);
221:             
222:         //Switch the type of menu
223:         switch($type) {
224:             case 'ul':
225:                 $menu = $this->ul($menu);
226:                 break;
227:             case 'nav':
228:                 $menu = $this->ul($menu, array('class' => 'nav'));
229:                 break;
230:         }
231:         
232:         return $menu;
233:     }
