 82:     public function admin_edit($id = NULL) {

 93:         } 
 94:         else
 95:             $this->request->data = $this->PostsCategory->find('first', array(
 96:                 'conditions'    => array('id' => $id),
 97:                 'fields'        => array('id', 'parent_id', 'title', 'slug', 'description')
 98:             ));
 99: 
100:         $this->set(array(
101:             'parents'           => $this->PostsCategory->generateTreeList(),

107:      * Delete posts category
108:      * @param string $id Posts category id
109:      * @throws NotFoundException
110:      */
111:     public function admin_delete($id = NULL) {
112:         $this->PostsCategory->id = $id;
113:         if(!$this->PostsCategory->exists())
114:             throw new NotFoundException(__d('me_cms', 'Invalid posts category'));
115:             
116:         $this->request->onlyAllow('post', 'delete');
117:         
118:         //Before deleting, it checks if the category has some posts
119:         if(!$this->PostsCategory->field('post_count')) {
120:             if($this->PostsCategory->delete())
121:                 $this->Session->flash(__d('me_cms', 'The posts category has been deleted'));
122:             else
123:                 $this->Session->flash(__d('me_cms', 'The posts category was not deleted'), 'error');
124:         }
125:         else
126:             $this->Session->flash(__d('me_cms', 'Before you delete this category, you have to delete its posts or assign them to another category'), 'error');
127:             
128:         $this->redirect(array('action' => 'index'));
129:     }

131:     /**
132:      * Gets the categories list, with the slug as key and the title as value.
133:      * This method works only with `requestAction()`.
134:      * @return array Categories list
135:      * @throws ForbiddenException
136:      */
137:     public function request_list() {
138:         //This method works only with "requestAction()"
139:         if(empty($this->request->params['requested']))
140:             throw new ForbiddenException();
141:         
142:         //Gets the categories
143:         $categories = $this->PostsCategory->find('active', array('fields' => array('id', 'slug', 'post_count')));
144:                 
145:         if(empty($categories))
146:             return array();
147:         
148:         //Gets the tree list
149:         $treeList = $this->PostsCategory->generateTreeList();
150:         
151:         $categoriesTmp = array();
152:         
153:         foreach($categories as $category) {
154:             //Changes the category titles, replacing them with the titles of the tree list and adding the "post_count" value
155:             $category['PostsCategory']['title'] = sprintf('%s (%s)', $treeList[$category['PostsCategory']['id']], $category['PostsCategory']['post_count']);
156:             
157:             //The new array has the slug as key and the title as value
158:             $categoriesTmp[$category['PostsCategory']['slug']] = $category['PostsCategory']['title'];
159:         }
160:         
161:         return $categoriesTmp;
162:     }
