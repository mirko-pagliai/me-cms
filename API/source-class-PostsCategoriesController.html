 72:             throw new NotFoundException(__d('me_cms', 'Invalid object'));
 73:             
 74:         $this->request->onlyAllow('post', 'delete');
 75:

 77:         if(!$this->PostsCategory->field('post_count')) {
 78:             if($this->PostsCategory->delete())
 79:                 $this->Session->flash(__d('me_cms', 'The posts category has been deleted'));
 80:             else
 81:                 $this->Session->flash(__d('me_cms', 'The posts category was not deleted'), 'error');
 82:         }
 83:         else
 84:             $this->Session->flash(__d('me_cms', 'Before you delete this category, you have to delete its posts or assign them to another category'), 'alert');

 94:     public function admin_edit($id = NULL) {

101:                 $this->redirect(array('action' => 'index'));

105:         } 
106:         else
107:             $this->request->data = $this->PostsCategory->findById($id, array('id', 'parent_id', 'title', 'slug', 'description'));
108: 
109:         $this->set(array(
110:             'parents'           => $this->PostsCategory->generateTreeList(),
111:             'title_for_layout'  => __d('me_cms', 'Edit posts category')

127:         array_walk($categories, function(&$v, $k, $treeList) {
128:             $v['PostsCategory']['title'] = $treeList[$v['PostsCategory']['id']];
129:         }, $this->PostsCategory->generateTreeList());
130:         
131:         $this->set(array(
132:             'categories'        => $categories,
133:             'title_for_layout'  => __d('me_cms', 'Posts categories'))
