262:      * Contact form
263:      * @uses config
264:      * @uses RecaptchaComponent::check()
265:      * @uses RecaptchaComponent::getError()
266:      */
267:     public function contact_form() {
268:         //Checks if the contact form is enabled
269:         if(!$this->config['frontend']['contact_form']) {
270:             $this->Session->flash(__d('me_cms', 'Disabled'), 'error');
271:             $this->redirect('/');
272:         }
273:         
274:         //Loads the `Contact` model
275:         $this->loadModel('MeCms.Contact');
276:             
277:         if($this->request->is('post') || $this->request->is('put')) {
278:             $this->Contact->set($data = $this->request->data);
279:             
280:             //Checks for reCAPTCHA, if requested
281:             if($this->config['security']['recaptcha'] && !$this->Recaptcha->check()) {
282:                 $this->Session->flash($this->Recaptcha->getError(), 'error');
283:             }
284:             elseif($this->Contact->validates()) {
285:                 //Loads the `Email` component               
286:                 $this->Email = $this->Components->load('MeCms.Email');
287:                 $this->Email->startup($this);
288:                 
289:                 //Sends email
290:                 $this->Email->from(array(
291:                     $data['Contact']['email'] => $full_name = sprintf('%s %s', $data['Contact']['first_name'], $data['Contact']['last_name'])
292:                 ));
293:                 $this->Email->to($this->config['email']['webmaster']);
294:                 $this->Email->subject(__d('me_cms', 'Email from %s', $this->config['main']['title']));
295:                 $this->Email->template('contact_form');
296:                 $this->Email->set(am(array('email' => $data['Contact']['email'], 'message' => $data['Contact']['message']), compact('full_name')));
297:                 $this->Email->send();
298:                 
299:                 $this->Session->flash(__d('me_cms', 'The email has been sent'));
300:                 $this->redirect('/');
301:             }
302:             else
303:                 $this->Session->flash(__d('me_cms', 'The email was not sent, try again'), 'error');
304:         }
305:         
306:         $this->set(array('title_for_layout' => __d('me_cms', 'Contact us')));
307:     }
