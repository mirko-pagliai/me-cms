 59:     /**
 60:      * Media browser with KCFinder
 61:      * @uses MeCms\Controller\Component\KcFinderComponent::checkKcfinder()
 62:      * @uses MeCms\Controller\Component\KcFinderComponent::checkFiles()
 63:      * @uses MeCms\Controller\Component\KcFinderComponent::getFilesPath()

 65:      * @uses MeCms\Controller\Component\KcFinderComponent::getTypes()
 66:      */
 67:     public function browser() {
 68:         //Loads the KcFinder component
 69:         $this->loadComponent('MeCms.KcFinder');
 70:         
 71:         //Checks for KCFinder
 72:         if(!$this->KcFinder->checkKcfinder()) {
 73:             $this->Flash->error(__d('me_cms', '{0} is not present into {1}', 'KCFinder', rtr($this->KcFinder->getKcfinderPath())));
 74:             $this->redirect(['_name' => 'dashboard']);
 75:         }
 76:         
 77:         //Checks for the files directory (`APP/webroot/files`)
 78:         if(!$this->KcFinder->checkFiles()) {
 79:             $this->Flash->error(__d('me_cms', 'The directory {0} is not readable or writable', rtr($this->KcFinder->getFilesPath())));
 80:             $this->redirect(['_name' => 'dashboard']);
 81:         }
 82:         
 83:         //Gets the supperted types from configuration
 84:         $types = $this->KcFinder->getTypes();
 85:         
 86:         //If there's only one type, it automatically sets the query value
 87:         if(!$this->request->query('type') && count($types) < 2)
 88:             $this->request->query['type'] = fk($types);
 89:         
 90:         //Gets the type from the query and the types from configuration
 91:         $type = $this->request->query('type');
 92:         
 93:         $locale = substr(\Cake\I18n\I18n::locale(), 0, 2);
 94:         
 95:         //Checks the type, then sets the KCFinder path
 96:         if($type && array_key_exists($type, $types))
 97:             $this->set('kcfinder', sprintf('%s/kcfinder/browse.php?lang=%s&type=%s', Router::url('/vendor', TRUE), empty($locale) ? 'en' : $locale, $type));
 98:         
 99:         $this->set('types', array_combine(array_keys($types), array_keys($types)));
100:     }
